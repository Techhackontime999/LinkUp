// Copy this code
const grid = document.querySelector('.grid.grid-cols-1');
const columns = grid ? grid.children : [];

console.log('=== LAYOUT DEBUG ===');
console.log('Grid found:', !!grid);
console.log('Total columns:', columns.length);

if (grid) {
    const style = window.getComputedStyle(grid);
    console.log('Display:', style.display);
    console.log('Grid template columns:', style.gridTemplateColumns);
    
    Array.from(columns).forEach((col, i) => {
        const colStyle = window.getComputedStyle(col);
        console.log(`Column ${i + 1}:`, {
            classes: col.className,
            gridColumn: colStyle.gridColumn,
            display: colStyle.display
        });
    });
}

// Check what's visible
const leftSidebar = document.querySelector('.lg\\:col-span-3:first-child');
const mainFeed = document.querySelector('.lg\\:col-span-6');
const rightSidebar = document.querySelector('.lg\\:col-span-3:last-child');

console.log('Left sidebar visible:', leftSidebar && window.getComputedStyle(leftSidebar).display !== 'none');
console.log('Main feed visible:', mainFeed && window.getComputedStyle(mainFeed).display !== 'none');
console.log('Right sidebar visible:', rightSidebar && window.getComputedStyle(rightSidebar).display !== 'none');
